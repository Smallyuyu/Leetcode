//頭尾增減
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        map<int,int> mp;
        for(int i = 0 ; i < k ; i++){
            mp[nums[i]]++;
        }
        vector<int> ans;
        ans.push_back(prev(mp.end())->first);
        for(int i = k ; i < nums.size() ; i++){
            mp[nums[i]]++;
            mp[nums[i - k]]--;
            if(!mp[nums[i - k]]) mp.erase(nums[i - k]);
            ans.push_back(prev(mp.end())->first);
        }
        return ans;
    }
};
